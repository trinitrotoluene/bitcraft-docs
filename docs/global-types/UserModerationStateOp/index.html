  user_moderation_policy: PermanentBlockLogin({ })
  | TemporaryBlockLogin({ })
  | BlockChat({ })
  | BlockConstruct({ }),
  created_time: {
    __timestamp_micros_since_unix_epoch__: I64

  duration_ms: U64
})
